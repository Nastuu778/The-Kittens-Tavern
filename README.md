**Структура/Назначения файлов**

├── README.md # Этот файл: описание проекта
├── requirements.txt # Зависимости Python (pip install -r requirements.txt)
├── .gitignore # Файлы, исключаемые из репозитория

**├── src/ # Исходный код игры**
│ ├── main.py # Точка входа: запуск класса Game
│ ├── settings.py # Глобальные константы: размеры, цвета, скорости
│ ├── game.py # Основной класс Game: игровой цикл, обработка событий
│ ├── player.py # Класс Player: управление персонажем, коллизии
│ ├── wall.py # Класс Wall: препятствия на локациях
│ ├── zone.py # Класс InteractionZone: зоны для диалогов с NPC
│ ├── location.py # Классы Location и TransitionZone: локации и переходы
│ ├── npc.py # Класс NPC: неигровые персонажи
│ └── dialog_system.py # Класс DialogSystem: загрузка и отображение диалогов


**├── data/ # Данные игры**
│ └── dialogs.json # Тексты диалогов для всех NPC (структура: NPC → реплики)
│
├── assets/ # Ресурсы игры (графика, звук)
│ ├── images/ # Спрайты: персонажи, фоны, UI-элементы
│ │ ├── characters/ # Изображения 4 персонажей
│ │ ├── locations/ # Фоны для 3 локаций
│ │ └── ui/ # Элементы интерфейса
│ └── sounds/ # Звуковые эффекты и музыка
